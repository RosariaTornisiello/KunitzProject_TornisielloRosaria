#creation of a pandas series that contains the UniProt ID and the class (1=postive, 0=negative)
import pandas as pd
import numpy as np
from Bio import SeqIO
from Bio import SearchIO

#domain series creation
domain = {}
for record in SeqIO.parse("./negatives_kunitz.fasta", "fasta"):
    ID_n = record.id.split("|")[1]
    domain[ID_n] = 0
for record in SeqIO.parse("./positives_clean_kunitz.fasta", "fasta"):
    ID_p = record.id.split("|")[1]
    domain[ID_p] = 1
domain = pd.Series(domain, dtype='category')
